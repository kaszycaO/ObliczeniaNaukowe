function ilorazyRoznicowe(x::Vector{Float64}, f::Vector{Float64})
    """ Funkcja obliczajca ilorazy rnicowe dla podanej funkcji i jej wartoci
    
    Dane wejciowe:
        x: wektor dugoci n + 1 zawierajcy wzy x0, ... , xn (x[1] = x0 itd.)
        f: wektor dugoci n + 1 zawierajcy wartoci funkcji w podanych wzach (x)
    
    Dane wyjciowe:
        fx: wektor dugoci n + 1 zawierajcy obliczone ilorazy rnicowe, w postaci:
            fx[1] = f[x0]; fx[2] = f[x0, x1] itd.       
    """     
    
    
    n = length(f)
    fx = copy(f)
    
    for i = 2:n
        for j = n:-1:i
            fx[j] = (fx[j] - fx[j-1]) / (x[j] - x[j-i+1])
        end
    end
    return fx
end

Opis¶

Powyszy algorytm to interpretacja nastpujcego wzoru sucego do wyliczania ilorazw rznicowych: $$f[x_i, x_{i+1}, \dots, x_{i+j}] = \frac{f[x_{i+1}, \dots, x_{i+j} ] - f[x_i, \dots, x_{i+j-1}]}{x_{i+j} - x_i}$$ Zakadajc, e $f[x_i] = f(x_i)$ oraz $f[x_i, x_j] = \frac{f(x_j) - f(x_i)}{x_j - x_i}$ w prosty sposb moemy skonstruowa tablic trjktn, atw do reprezentacji graficznej. Analizujc jej konstrukcj, mona zauway, e interesujcy nas wynik znajduje si w jej pierwszym rzdzie, wic algorytm nie musi korzysta z macierzy (zgodnie z treci zadania).

Na pocztku, tworzona jest kopia wektora f (fx), ktra bdzie przechowywaa ostateczny wynik. Nastpnie swoje dziaanie rozpoczynaj dwie ptle: zewntrzna, odpowiedzialna za nawigacj midzy kolejnymi kolumnami tablicy trjktnej; oraz wewntrzna, ktra uzupenia wartoci w kolejnych wierszach kadej kolumny, wdrujc od ostatniego wiersza (dziki czemu nie grozi nam utrata potrzebnych wartoci). Z kad iteracj kolejne kolumny s nadpisywane nowymi wartociami, zgodnie z podanym wczeniej wzorem. Obliczenia s zatem wykonywane bez uycia macierzy. Bo zakoczonych iteracjach, zwracany jest wektor reprezentujcy pierwszy rzd tablicy trjktnej.

function warNewton(x::Vector{Float64}, fx::Vector{Float64}, t::Float64)    
    """ Funkcja obliczajca warto wielomianu interpolacyjnego w punkcie x = t,
        za pomoc uoglnionego algorytmu Hornera.
        
        Dane wejciowe:
            x: wektor dugoci n + 1 zawierajcy wzy x0, ... , xn (x[1] = x0 itd.)
            fx: wektor dugoci n + 1 zawierajcy ilorazy rnicowe, w postaci:
                fx[1] = f[x0]; fx[2] = f[x0, x1] itd.
            t: punkt, w ktrym zostanie obliczona warto wielomianu (Float64)
    
        Dane wyjciowe:
            nt: warto wielomianu w punkcie t (Float64)
    
    """ 
    n = length(x)
    nt = fx[n]
    
    for k = n - 1:-1:1
        nt = fx[k] + (t-x[k]) * nt
    end
    
    return nt
    
end

Opis¶

Wielomian interpolacyjny Newtona moemy przedstawi nastpujco: $$N_n(x) = f[x_0] + f[x_0, x_1](x - x_0) + \dots + f[x_0, x_1, \dots, x_n](x - x_0)(x - x_1) \dots (x - x_{n-1})$$ Taka reprezentacja pozwala na obliczenie wartoci tego wielomianu, w dowolnym punkcie, za pomoc nastpujcych wzorw (uoglniony algorytm Hornera):

1) $w_n(x) = f[x_0, x_1, \dots , x_n]$

2) $w_k(x) = f[x_0, x_1, \dots , x_k] + (x - x_k)w_{k+1}(x)$, gdzie $k=n-1, \dots, 0$

3) $N_n(x) = w_0(x)$

Wzory te s wykorzystane w podanym algorytmie.

Na pocztku tworzona jest zmienna $nt$, ktra bdzie przechowywaa ostateczny wynik. Poniewa ptla bdzie iterowaa po elementach fx od koca (k = n-1), warto zmiennej nt ustawiamy na ostatni warto przechowywan w wektorze fx. Kolejne iteracje ptli to de facto kolejne kroki uoglnionego algorytmu Hornera (w postaci wzoru 2)). Gdy ptla zakoczy swoje dziaanie, zwracany jest wynik, przechowywany przez zmienn nt.

Koszt wykonania tego algorytmu to $O(n)$ - jedna ptla wykonujca si n-1 razy.

function naturalna(x::Vector{Float64}, fx::Vector{Float64})
    """ Funkcja obliczajca wspczynniki wielomianu interpolacyjnego w jego postaci naturalnej 
        
        Dane wejciowe:
            x: wektor dugoci n + 1 zawierajcy wzy x0, ... , xn (x[1] = x0 itd.)
            fx: wektor dugoci n + 1 zawierajcy ilorazy rnicowe, w postaci:
                fx[1] = f[x0]; fx[2] = f[x0, x1] itd.
        
        Dane wyjciowe:
            a: wekotor dugoci n + 1 zawierajcy obliczone wspczynniki postaci naturalnej, w postaci:
                a[1] = a0; a[2] = a1; a[n+1] = an, gdzie w(x) = anx^n + a(n-1)x^(n-1)+...+a0 itd.
    """ 
    
    n = length(x)
    a = Vector{Float64}(undef, n)
    a[n] = fx[n]
    
    for k = n - 1:-1:1
        a[k] = fx[k] - x[k] * a[k+1]
        for j = (k + 1):(n - 1)
           a[j] = a[j] - x[k] * a[j + 1] 
        end
    end
    return a
end

Opis¶

Obliczenie wspczynnikw wielomianu dla jego postaci naturalnej rwnie opiera si na wykorzystaniu wzorw, ktre podaem w poprzednim zadaniu. Sama konstrukcja algorytmu jest bardzo podobna, z wyjtkiem paru istotnych rnic, ktre objani poniej.

Na pocztku dziaania, tworzony jest wektor a, ktry bdzie przechowywa obliczone wspczynniki. Analogicznie jak w zadaniu 2., obliczenia bd wykonywane "od koca", dlatego $a[n] = fx[n]$. Nastpnie pojawiaj si dwie ptle, pierwsza, podobnie jak w zadaniu 2., oblicza "skadowe" wielomiany (zgodnie z podanymi wzorami), jednak tym razem s one osobno zapisywane w wektorze a. Ptla wewntrzna ma za zadanie doprowadzi utworzony w ptli zewntrznej, "skadowy" wielomian do postaci naturalnej. Kiedy ptla zakoczy dziaanie (przeiteruje po kadym elemencie fx), zostaje zwrcony wektor a, ze wspczynnikami dla postaci naturalnej wielomianu interpolacyjnego.

Koszt wykonania tego algorytmu to $O(n^2)$ - ptla zewntrzna wykonuje si $n-1$ razy, a ptla wewntrzna co najwyej $n$ razy. W przypdaku pesymistycznym mamy zatem: $O(n * (n - 1)) = O(n^2)$

using Plots

function rysujNnfx(f, a::Float64, b::Float64, n::Int)
    """ Funkcja przeprowadzajca interpolacj podanej funkcji w podanym przedziale oraz
        rysujca podanfunkcj i jej wielomian interpolacyjny.
    
        Dane wejciowe:
            f: funkcja zadana jako anonimowa
            a,b: przedzia interpolacji
            n: stopie wielomianu interpolacyjnego
    
        Dane wyjciowe:
            BRAK: funkcja rysuje funkcj oraz wielomian interpolacyjny
    """
    
    accuracy = 20
    n_max = n + 1 
    
    x = Vector{Float64}(undef, n_max)
    y = Vector{Float64}(undef, n_max)
    f_x = Vector{Float64}(undef, n_max) 
    
    plot_x = Vector{Float64}(undef, n_max*accuracy) 
    plot_y = Vector{Float64}(undef, n_max*accuracy) 
    plot_ip = Vector{Float64}(undef, n_max*accuracy) 
    
    
    h = (b - a) / n
    kh = Float64(0.0)


    for i = 1:n_max
        x[i] = a + kh
        y[i] = f(x[i])
        kh += h
    end
    
    f_x = ilorazyRoznicowe(x, y)
    
    kh = Float64(0.0)
    n_max *= accuracy
    h = (b - a) / (n_max - 1)

    for i = 1:n_max
        plot_x[i] = a + kh
        plot_y[i] = f(plot_x[i])
        plot_ip[i] = warNewton(x, f_x, plot_x[i])
        kh += h
    end

    title = "Interpolacja dla n = " * string(n)
    plot(plot_x, [plot_y, plot_ip], label=["f(x)" "w(x)"], lw=2, title=title)

end

Opis¶

Wykonanie interpolacji funkcji opiera sina algorytmach, ktre zostay zaimplementowane wczeniej (ilorazy rnicowe (zad1), obliczanie wartoci wielomianu w punkcie w postaci Newtona (zad2)). Elementem w algorytmie, ktry zosta dodany to wyznaczenie rwnoodlegych od siebie wzw, ktre wykonywane jest w pierwszej ptli. Wie si to z nastpujcymi wzorami: $x_k = a + kh, h = \frac{b - a}{n}$ dla $k = 0, 1, \dots, n$.

Na pocztku ustalana jest dokadno rysowanych wykresw (mnonik accuracy, ktry zwiksza liczb wzw, na ktrych zostanie wywaana funkcja warNewton. Po zdefiniowaniu potrzebnych zmiennych i wektorw, wyznaczane s wzy: ustalana jest odlego h oraz inicjalizowany wspczynnik $kh$. Nastpnie w ptli obliczane jest $n_{max}$ wzw oraz wartoci podanej funkcji (f), w tych wzach.Tak przygotowane dane s przekazywane do funkcji ilorazyRoznicowe. W kolejnym kroku, bd liczone wartoci wielomianu interpolacyjnego, za pomoc funkcji warNewton. Przed wykonaniem ptli, wykonujcej to zadanie, ponownie wyznaczane s wzy (tym razem bdzie ich wicej, tak eby wykres funkcji by dokadniejszy). Otrzymane w ten sposb dane (plot_x, plot_y oraz wartoci po przeprowadzonej interpolacji - plot_ip) przedstawiane s na wykresie.
